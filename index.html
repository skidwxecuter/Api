import express from 'express'; import cors from 'cors'; import bodyParser from 'body-parser'; import fs from 'fs'; import path from 'path'; const app = express(); app.use(cors()); app.use(bodyParser.json()); app.use(express.static('public')); // Ensure /public exists const publicDir = path.join(process.cwd(), 'public'); if (!fs.existsSync(publicDir)) { fs.mkdirSync(publicDir); } // Ensure redirects.json exists const dbPath = path.join(publicDir, 'redirects.json'); if (!fs.existsSync(dbPath)) { fs.writeFileSync(dbPath, JSON.stringify([])); } // Create redirect endpoint app.post('/create-redirect', (req, res) => { const { targetUrl, siteName } = req.body; if (!targetUrl || !siteName) { return res.status(400).json({ error: 'Missing targetUrl or siteName' }); } const html = ` Redirecting...

If you are not redirected, click here.

`.trim(); const filePath = path.join(publicDir, `${siteName}.html`); fs.writeFile(filePath, html, (err) => { if (err) { console.error(err); return res.status(500).json({ error: 'Failed to save file' }); } // Save to redirects.json const list = JSON.parse(fs.readFileSync(dbPath, 'utf-8')); list.unshift({ name: siteName, url: `/${siteName}.html`, created: new Date().toISOString() }); fs.writeFileSync(dbPath, JSON.stringify(list, null, 2)); return res.json({ url: `${req.protocol}://${req.get('host')}/${siteName}.html` }); }); }); // Return recent redirects app.get('/list', (req, res) => { const list = JSON.parse(fs.readFileSync(dbPath, 'utf-8')); res.json(list); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
